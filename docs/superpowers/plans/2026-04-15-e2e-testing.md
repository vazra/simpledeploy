# E2E Testing Suite Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a comprehensive Playwright E2E test suite that covers every SimpleDeploy feature, runs against a real server+Docker, and serves as release quality gate.

**Architecture:** Single Playwright project in `e2e/`. Tests build the Go binary, start a server with temp config, and run serially in user-session order. State builds up across tests (setup -> deploy -> actions -> cleanup). Hash-based SPA routing (`#/path`).

**Tech Stack:** Playwright, Node.js, SimpleDeploy Go binary, Docker

### Task 2: Server Lifecycle Helper

**Files:**
- Create: `e2e/helpers/server.js`

- [ ] **Step 1: Create server.js**

This module builds the binary, generates a test config, starts the server, waits for health, and provides cleanup.

```javascript
import { execFileSync, spawn } from 'child_process';
import { mkdtempSync, writeFileSync, rmSync, existsSync, readFileSync, createWriteStream } from 'fs';
import { join } from 'path';
import { tmpdir } from 'os';
import net from 'net';

const ROOT = join(import.meta.dirname, '..', '..');

function getAvailablePort() {
  return new Promise((resolve, reject) => {
    const srv = net.createServer();
    srv.listen(0, () => {
      const port = srv.address().port;
      srv.close(() => resolve(port));
    });
    srv.on('error', reject);
  });
}

export async function buildBinary() {
  console.log('[e2e] Building SimpleDeploy binary...');
  execFileSync('make', ['build'], { cwd: ROOT, stdio: 'inherit' });
  const bin = join(ROOT, 'bin', 'simpledeploy');
  if (!existsSync(bin)) throw new Error('Binary not found after build');
  return bin;
}

export async function startServer(binPath) {
  const port = await getAvailablePort();
  const dataDir = mkdtempSync(join(tmpdir(), 'sd-e2e-data-'));
  const appsDir = mkdtempSync(join(tmpdir(), 'sd-e2e-apps-'));
  const configPath = join(dataDir, 'config.yml');
  const logPath = join(dataDir, 'server.log');

  const config = [
    `data_dir: "${dataDir}"`,
    `apps_dir: "${appsDir}"`,
    `listen_addr: ":0"`,
    `management_port: ${port}`,
    `master_secret: "e2e-test-secret-key-32bytes!!"`,
    `log_buffer_size: 100`,
    `tls:`,
    `  mode: "off"`,
  ].join('\n');

  writeFileSync(configPath, config);

  console.log(`[e2e] Starting server on port ${port}...`);
  console.log(`[e2e] Data dir: ${dataDir}`);
  console.log(`[e2e] Apps dir: ${appsDir}`);

  const logStream = createWriteStream(logPath);
  const proc = spawn(binPath, ['serve', '--config', configPath], {
    cwd: ROOT,
    stdio: ['ignore', 'pipe', 'pipe'],
    env: { ...process.env },
  });

  proc.stdout.pipe(logStream);
  proc.stderr.pipe(logStream);

  proc.on('error', (err) => {
    console.error('[e2e] Server process error:', err);
  });

  // Wait for health endpoint
  const baseURL = `http://localhost:${port}`;
  const deadline = Date.now() + 30_000;
  while (Date.now() < deadline) {
    try {
      const res = await fetch(`${baseURL}/api/health`);
      if (res.ok) {
        console.log(`[e2e] Server healthy at ${baseURL}`);
        return { proc, port, dataDir, appsDir, configPath, logPath, baseURL };
      }
    } catch {}
    await new Promise((r) => setTimeout(r, 300));
  }

  // If we get here, server didn't start
  proc.kill('SIGKILL');
  const logs = existsSync(logPath) ? readFileSync(logPath, 'utf-8') : '(no logs)';
  throw new Error(`Server failed to start within 30s.\nLogs:\n${logs}`);
}

export function stopServer(server) {
  if (!server) return;
  console.log('[e2e] Stopping server...');
  try {
    server.proc.kill('SIGTERM');
  } catch {}
  // Cleanup temp dirs
  for (const dir of [server.dataDir, server.appsDir]) {
    try {
      rmSync(dir, { recursive: true, force: true });
    } catch {}
  }
  try {
    rmSync(server.configPath, { force: true });
  } catch {}
}
```

```javascript
import { readFileSync, rmSync, existsSync } from 'fs';
import { join } from 'path';

const STATE_FILE = join(import.meta.dirname, '.e2e-state.json');

export default async function globalTeardown() {
  if (!existsSync(STATE_FILE)) return;

  const state = JSON.parse(readFileSync(STATE_FILE, 'utf-8'));

  // Kill server
  try {
    process.kill(state.pid, 'SIGTERM');
    // Wait briefly for graceful shutdown
    await new Promise((r) => setTimeout(r, 2000));
    try { process.kill(state.pid, 'SIGKILL'); } catch {}
  } catch {}

  // Cleanup
  for (const dir of [state.dataDir, state.appsDir]) {
    try { rmSync(dir, { recursive: true, force: true }); } catch {}
  }
  try { rmSync(state.configPath, { force: true }); } catch {}
  try { rmSync(STATE_FILE, { force: true }); } catch {}

  console.log('[e2e] Cleanup complete');
}
```

```javascript
import { getState } from './auth.js';

let sessionCookie = null;

export async function apiRequest(method, path, body) {
  const state = getState();
  const opts = {
    method,
    headers: { 'Content-Type': 'application/json' },
  };
  if (sessionCookie) {
    opts.headers['Cookie'] = sessionCookie;
  }
  if (body) {
    opts.body = JSON.stringify(body);
  }
  const res = await fetch(`${state.baseURL}${path}`, opts);
  // Capture session cookie from login response
  const setCookie = res.headers.get('set-cookie');
  if (setCookie && setCookie.includes('session=')) {
    sessionCookie = setCookie.split(';')[0];
  }
  const text = await res.text();
  let data;
  try { data = JSON.parse(text); } catch { data = text; }
  return { status: res.status, data, ok: res.ok };
}

export async function apiLogin(username, password) {
  return apiRequest('POST', '/api/auth/login', { username, password });
}

export async function waitForAppStatus(slug, status, timeoutMs = 60_000) {
  const deadline = Date.now() + timeoutMs;
  while (Date.now() < deadline) {
    const res = await apiRequest('GET', `/api/apps/${slug}`);
    if (res.ok && res.data.status === status) return res.data;
    await new Promise((r) => setTimeout(r, 1_000));
  }
  throw new Error(`App ${slug} did not reach status "${status}" within ${timeoutMs}ms`);
}
```
